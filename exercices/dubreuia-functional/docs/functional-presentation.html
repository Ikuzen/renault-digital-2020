List<String> imperative(List<String> elements) {
  List<String> lowerCases = new ArrayList<>();
  for (String element : elements) {
    lowerCases.add(element.toLowerCase());
  }
  return lowerCases;
}

List<String> functional(List<String> elements) {
  return elements.stream()                   // (1) stream start
      .map(element -> element.toLowerCase()) // (2) lazy operation(s)
      .collect(Collectors.toList());         // (3) terminal operation
}
